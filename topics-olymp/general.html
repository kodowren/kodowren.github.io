                            Program sorag_1;
                            const nMax=100;
                            var a:array[1..nMax] of integer;
                            begin
                                //Massiwe random bahalar bermek
                                var n := ReadInteger('Massiwin element sanyny girizin: ');
                                for var i:=1 to n do a[i] := random(-10, 10);                               
                                //Massiwi cap etmek
                                for var i:=1 to n do write(a[i]:4);
                                writeln;                       
                                //Massiwi (artyan tertipde) tertiplesdirmek - Bubble Sort
                                for var i:=1 to n-1 do
                                    for var j:=n downto i+1 do
                                    if a[j] < a[j-1] then swap(a[j],a[j-1]);                             
                                //Kopeltmek hasyly in uly bolan uc element
                                writeln(a[n]:3, a[n-1]:3, a[n-2]:3);
                            end.
